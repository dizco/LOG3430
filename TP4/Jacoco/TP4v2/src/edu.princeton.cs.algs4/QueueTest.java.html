package edu.princeton.cs.algs4;

import static org.junit.Assert.*;

import java.util.NoSuchElementException;

import org.junit.After;
import org.junit.Before;
import org.junit.Test;


public class QueueTest {

	Queue q;
	
	@Before  
    public void setUp()
    {
		q = new Queue();
    }
	
	@Test
	public void testIsEmptyVide() {
		q.isEmpty();
		assertEmpty();
	}
	
	@Test
	public void testToStringVide() {
		q.toString();
		assertEmpty();
	}
	
	@Test
	public void testSizeVide() {
		q.size();
		assertEmpty();
	}
	
	@Test
	public void testIteratorVide() {
		q.iterator();
		assertEmpty();
	}
	
	@Test
	public void testDequeueVide() {
		try {
			q.dequeue();
			fail("Dequeue ne devrait pas fonctionner avec une queue vide");
		}
		catch (NoSuchElementException e) {
			//Catch
		}
	}
	
	@Test
	public void testPeekVide() {
		try {
			q.peek();
			fail("Peek ne devrait pas fonctionner avec une queue vide");
		}
		catch (NoSuchElementException e) {
			//Catch
		}
	}
	
	
	@Test
	public void testEnqueueVide() {
		q.enqueue("item");
		assertNotEmpty();
	}
	
	@Test
	public void testDequeueNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.dequeue();
		assertEmpty();
	}
	
	@Test
	public void testToStringNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.toString();
		assertNotEmpty();
	}
	
	@Test
	public void testSizeNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.size();
		assertNotEmpty();
	}
	
	@Test
	public void testIteratorNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.iterator();
		assertNotEmpty();
	}
	
	@Test
	public void testEnqueueNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.enqueue("item2");
		assertNotEmpty();
	}
	
	@Test
	public void testPeekNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.peek();
		assertNotEmpty();
	}
	
	@Test
	public void testIsEmptyNonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.isEmpty();
		assertNotEmpty();
	}
	
	@Test
	public void testDequeue2NonVide() {
		q.enqueue("item");
		assertNotEmpty();
		q.enqueue("item2");
		q.dequeue();
		assertNotEmpty();
	}
	
	
	public void assertEmpty() {
		assertEquals("Queue devrait etre vide", 0, q.size());
	}
	
	public void assertNotEmpty() {
		assertTrue("Queue ne devrait pas etre vide", q.size() > 0);
	}

}
